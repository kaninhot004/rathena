-	script	Events	-1,{
	end;
	
OnPCLoginEvent:
	if(!isAutoLootOff)
	autoloot();
	if(IsWeekend())
	announce "พิเศษวันหยุด | ราคา Buff ถูกลง",bc_self,0xd598f0,0,23;

	donpcevent "เปลี่ยน Save Point::OnRefreshOnlineAmount";
	
	if(!IsCustomMap(strcharinfo(3)))
	addtimer 1000,strnpcinfo(0) + "::OnLoginWarp";
	
	end;
	
OnLoginWarp:
	warp "Save",0,0;
	end;
	
OnPCLogoutEvent:
	donpcevent "เปลี่ยน Save Point::OnRefreshOnlineAmount";
	end;
	
OnPCLoadMapEvent:
	if(strcharinfo(3) == "prontera"){
		if(bonusDebuff)
		ClearTheBoxDebuff();
		
		atcommand "@cleartimetick";
		
		if(@isOnlyGodEnter)
		@isOnlyGodEnter = 0;
		
		.@oldInstanceId = instance_id(IM_CHAR);
		if(.@oldInstanceId)
		instance_destroy .@oldInstanceId;
		.@oldInstanceId = instance_id(IM_PARTY);
		if(.@oldInstanceId)
		instance_destroy .@oldInstanceId;
	}
	end;
	
OnInit:
	bindatcmd "load",strnpcinfo(3) + "::OnLoad";
	bindatcmd "lb",strnpcinfo(3) + "::OnLastFollowPlayerBan",99,99;
	bindatcmd "ll",strnpcinfo(3) + "::OnLastFollowPlayerLoad",99,99;
	bindatcmd "ml",strnpcinfo(3) + "::OnMapLoad",99,99;
	
	addrid(0,1);
	if(!IsCustomMap(strcharinfo(3)))
	warp "Save",0,0;
	end;
	
OnMapLoad:
	mapwarp strcharinfo(3),"prontera",0,0;
	end;
	
OnLastFollowPlayerLoad:
	if(followCID)
	atcommand "#go " + strcharinfo(0,followCID) + " 0";
	end;
	
OnLastFollowPlayerBan:
	if(followCID)
	atcommand "@ban +30d " + strcharinfo(0,followCID);
	end;

OnLoad:
	.@isSkip = (@isFollow || (strcharinfo(3) == "prontera"));
	if(@isFollow){
		setpcblock PCBLOCK_ALL,false;
		atcommand "@follow";
		atcommand "@speed 200";
		ClearTheBoxDebuff();
		dispbottom "หยุดติดตาม " + strcharinfo(0,followCID);
		@isFollow = 0;
	}
	if(!.@isSkip){
		specialeffect2 35;
		progressbar "",(strcharinfo(3) == "guild_vs1") ? 4 : 2;
	}
	warp "Save",0,0;
	end;
}
