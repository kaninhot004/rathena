OnPCLoginEvent:
	if(!isResetSkill){
		isResetSkill = 1;
		atcommand "@resetskill";
	}
	Zeny = 1000000;
	if(!isJobSelected){
		atcommand "@clearinventory";
	}
	if(!IsSafeMap())
	{
		addtimer 1000,"Events::OnWarpBack";
	}
	end;
	
OnPCLogoutEvent:
	ClearAddTimer();
	end;
	
OnPCLoadMapEvent:
	ClearAddTimer();
	end;
	
OnInit:
	bindatcmd "load",strnpcinfo(3) + "::OnLoad";
	addrid(0);
	if(!IsSafeMap())
	{
		warp "Save",0,0;
		ClearAddTimer();
	}
	end;

OnLoad:
	PaWButterflyWing();
	end;
	
OnWarpBack:
	warp "Save",0,0;
	ClearAddTimer();
	end;
}
