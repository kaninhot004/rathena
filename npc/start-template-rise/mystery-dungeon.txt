-	script	MysteryDungeon	-1,{
	end;
	
OnMinute00:
OnMinute10:
OnMinute20:
OnMinute30:
OnMinute40:
OnMinute50:
	.@isTimeRefresh = 1;
	$mysteryDungeonCD = 0;
	L_RefreshMonsterId:
	if ($mysteryDungeonCD)
	return;
	$mysteryDungeonCD = 1;
	for(.@i = 0; .@i < getarraysize($mysteryDungeonMapName$); .@i++){
		if(.@isTimeRefresh)
		killmonsterall $mysteryDungeonMapName$[.@i];
		deletearray getd("$mysteryDungeonMID" + .@i + "[0]"),getarraysize(getd("$mysteryDungeonMID" + .@i));
		setarray getd("$mysteryDungeonMID" + .@i + "[0]"),$attackableMonsterIds[rand(getarraysize($attackableMonsterIds))],$attackableMonsterIds[rand(getarraysize($attackableMonsterIds))],$attackableMonsterIds[rand(getarraysize($attackableMonsterIds))],$attackableMonsterIds[rand(getarraysize($attackableMonsterIds))];
		set getd("$mysteryDungeonMVP" + .@i),$mvpIds[rand(getarraysize($mvpIds))];
		set getd("$isMysteryDungeonBossDead" + .@i),0;
		if(.@isTimeRefresh)
		callsub OnSpawnMore,.@i,$mysteryDungeonSpawnAmount / ((.@i % 2 == 0) ? 2 : 1);
	}
	if(.@isTimeRefresh)
	end;
	return;


OnSpawnMore:
	.@index = getarg(0,0);
	.@amount = getarg(1,0);
	.@isNonBoss = getarg(2,1);
	.@isBoss = getarg(3,1);
	if(.@isNonBoss){
		freeloop(1);
		for(.@i = 0; .@i < .@amount; .@i++)
		monster $mysteryDungeonMapName$[.@index],0,0,"--ja--",getd("$mysteryDungeonMID" + .@index + "[" + rand(0,3) + "]"),1,strnpcinfo(0) + "::OnMonsterDead" + .@index;
		freeloop(0);
	}
	if(.@isBoss && !getd("$isMysteryDungeonBossDead" + .@index))
	monster $mysteryDungeonMapName$[.@index],0,0,"--ja--",getd("$mysteryDungeonMVP" + .@index),1,strnpcinfo(0) + "::OnBossDead" + .@index;
	return;

OnMonsterDead0: callsub OnGetReward,0; callsub OnSpawnMore,0,1,1,0; end;
OnBossDead0: callsub OnGetReward,1; $isMysteryDungeonBossDead0 = 1; end;
OnMonsterDead1: callsub OnGetReward,0; callsub OnSpawnMore,1,1,1,0; end;
OnBossDead1: callsub OnGetReward,1; $isMysteryDungeonBossDead1 = 1; end;
OnMonsterDead2: callsub OnGetReward,0; callsub OnSpawnMore,2,1,1,0; end;
OnBossDead2: callsub OnGetReward,1; $isMysteryDungeonBossDead2 = 1; end;
OnMonsterDead3: callsub OnGetReward,0; callsub OnSpawnMore,3,1,1,0; end;
OnBossDead3: callsub OnGetReward,1; $isMysteryDungeonBossDead3 = 1; end;
OnMonsterDead4: callsub OnGetReward,0; callsub OnSpawnMore,4,1,1,0; end;
OnBossDead4: callsub OnGetReward,1; $isMysteryDungeonBossDead4 = 1; end;
OnMonsterDead5: callsub OnGetReward,0; callsub OnSpawnMore,5,1,1,0; end;
OnBossDead5: callsub OnGetReward,1; $isMysteryDungeonBossDead5 = 1; end;
OnMonsterDead6: callsub OnGetReward,0; callsub OnSpawnMore,6,1,1,0; end;
OnBossDead6: callsub OnGetReward,1; $isMysteryDungeonBossDead6 = 1; end;
OnMonsterDead7: callsub OnGetReward,0; callsub OnSpawnMore,7,1,1,0; end;
OnBossDead7: callsub OnGetReward,1; $isMysteryDungeonBossDead7 = 1; end;
OnMonsterDead8: callsub OnGetReward,0; callsub OnSpawnMore,8,1,1,0; end;
OnBossDead8: callsub OnGetReward,1; $isMysteryDungeonBossDead8 = 1; end;
OnMonsterDead9: callsub OnGetReward,0; callsub OnSpawnMore,9,1,1,0; end;
OnBossDead9: callsub OnGetReward,1; $isMysteryDungeonBossDead9 = 1; end;

OnGetReward:
	.@isBoss = getarg(0,0);
	if(rand(100) < (.@isBoss ? 100 : 1))
	getitem RISE_STONE_ITEM_ID,1;
	return;
	
OnSetup:
	if(!$isMysteryDungeonInit){
		$isMysteryDungeonInit = 1;
		
		callsub L_RefreshMonsterId;
		for(.i = 0; .i < getarraysize($mysteryDungeonMapName$); .i++)
		callsub OnSpawnMore,.i,$mysteryDungeonSpawnAmount;
	}
	end;

OnInit:
	$mysteryDungeonSpawnAmount = 150;
	
	deletearray $mysteryDungeonMapName$[0],getarraysize($mysteryDungeonMapName$);
	setarray $mysteryDungeonMapName$[0],"ordeal_1-1","ordeal_1-2","ordeal_2-1","ordeal_2-2","ordeal_3-1","ordeal_3-2";
	
	if($isMysteryDungeonInit){
		callsub L_RefreshMonsterId;
		for(.i = 0; .i < getarraysize($mysteryDungeonMapName$); .i++)
		callsub OnSpawnMore,.i,$mysteryDungeonSpawnAmount;
	}
	end;
}
