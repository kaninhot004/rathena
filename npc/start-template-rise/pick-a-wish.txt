	traitstatusup2 bWis,traitWis[.@index];
	traitstatusup2 bSpl,traitSpl[.@index];
	traitstatusup2 bCon,traitCon[.@index];

		//skill $skillIds[.@skillIndex],$skillMaxLvs[.@skillIndex],3;
		bonus_script "skill " + $skillIds[.@skillIndex] + "," + $skillMaxLvs[.@skillIndex] + ",3;",INFINITE_TICK,512,1;
	}
	
	// Equipment
	for(.@r = 0; .@r <= 20; .@r++)
	{
		.@itemId = getd("itemId" + .@index + "[" + .@r + "]");
		.@refine = getd("refine" + .@index + "[" + .@r + "]");
		.@card1 = getd("card" + .@index + "_" + .@r + "[0]");
		.@card2 = getd("card" + .@index + "_" + .@r + "[1]");
		.@card3 = getd("card" + .@index + "_" + .@r + "[2]");
		.@card4 = getd("card" + .@index + "_" + .@r + "[3]");
		.@enchantment1 = getd("enchantment" + .@index + "_" + .@r + "[0]");
		.@enchantment2 = getd("enchantment" + .@index + "_" + .@r + "[1]");
		.@enchantment3 = getd("enchantment" + .@index + "_" + .@r + "[2]");
		.@enchantment4 = getd("enchantment" + .@index + "_" + .@r + "[3]");
		.@slot1 = (.@card1 > 0) ? .@card1 : .@enchantment1;
		.@slot2 = (.@card2 > 0) ? .@card2 : .@enchantment2;
		.@slot3 = (.@card3 > 0) ? .@card3 : .@enchantment3;
		.@slot4 = (.@card4 > 0) ? .@card4 : .@enchantment4;
		.@grade = getd("grade" + .@index + "[" + .@r + "]");
		setarray .@randomOptionIds[0],getd("randomOption" + .@index + "_" + .@r + "[0]"),getd("randomOption" + .@index + "_" + .@r + "[1]"),getd("randomOption" + .@index + "_" + .@r + "[2]"),getd("randomOption" + .@index + "_" + .@r + "[3]"),getd("randomOption" + .@index + "_" + .@r + "[4]");
		setarray .@randomOptionValues[0],1,1,1,1,1;
		setarray .@randomOptionParams[0],0,0,0,0,0;
		if(.@itemId > 0)
		{
			getitem4 .@itemId,1,1,.@refine,0,.@slot1,.@slot2,.@slot3,.@slot4,.@grade,.@randomOptionIds,.@randomOptionValues,.@randomOptionParams;
			equip .@itemId;
		}
	}
	
	// Run equip again to prevent Accessory not equipped
	for(.@r = 0; .@r <= 20; .@r++)
	{
		.@itemId = getd("itemId" + .@index + "[" + .@r + "]");
		if(.@itemId > 0)
		{
			equip .@itemId;
		}
	}
	
	freeloop(0);
	
	FullHeal();
	
	dispbottom "รับพร Set ที่ " + (.@index + 1) + " เรียบร้อยแล้ว!",0x33ffa5;
	
	end;
	
OnInit:
	waitingroom "รับ" + strnpcinfo(1),0;
	end;
